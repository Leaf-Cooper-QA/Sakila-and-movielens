SELECT last_name FROM actor GROUP BY last_name HAVING COUNT(last_name) = 1 ORDER BY last_name DESC;
SELECT last_name,COUNT(last_name) FROM actor GROUP BY last_name HAVING COUNT(last_name) > 1 ORDER BY COUNT(last_name) DESC;
SELECT a.first_name,a.last_name,COUNT(a.actor_id) FROM actor AS a JOIN film_actor AS fa ON fa.actor_id=a.actor_id GROUP BY fa.actor_id ORDER BY COUNT(a.actor_id) DESC LIMIT 1;
SELECT r.return_date FROM rental AS r JOIN film AS f ON r.inventory_id=f.film_id WHERE f.title='ACADEMY DINOSAUR';
SELECT AVG(length) FROM film;
SELECT category, AVG(length) FROM film_list GROUP BY category ORDER BY AVG(length) DESC;
